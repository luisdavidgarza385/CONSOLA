Instrucciones para tu Archivo Excel

Copia el siguiente código y pégalo en el módulo `ThisWorkbook` de tu archivo Excel (.xlsm) para que se actualice automáticamente al abrirse.


Private Sub Workbook_Open()
    Dim http As Object
    Dim adodbStream As Object
    Dim url As String
    Dim savePath As String
    
    ' La URL de tu consola web (Asegúrate de que la IP o dominio es correcto cuando la publiques)
    url = "http://localhost:3000/api/download/latest"
    
    ' Dónde se guardará la DLL (en este caso, en la misma carpeta que el Excel)
    savePath = ThisWorkbook.Path & "\actualizado.dll"
    
    Set http = CreateObject("MSXML2.XMLHTTP")
    
    ' Petición para descargar el archivo
    http.Open "GET", url, False
    http.send
    
    If http.Status = 200 Then
        Set adodbStream = CreateObject("ADODB.Stream")
        adodbStream.Open
        adodbStream.Type = 1 ' adTypeBinary
        adodbStream.Write http.responseBody
        adodbStream.Position = 0
        adodbStream.SaveToFile savePath, 2 ' adSaveCreateOverWrite
        adodbStream.Close
        MsgBox "DLL actualizada exitosamente.", vbInformation, "Actualización"
    Else
        MsgBox "No se pudo actualizar la DLL. Error " & http.Status, vbExclamation, "Error"
    End If
    
    Set http = Nothing
    Set adodbStream = Nothing
End Sub
